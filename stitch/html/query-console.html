SELECT 
    e.employee_id, 
    e.first_name, 
    e.last_name, 
    e.role,
    p.efficiency_rate,
    p.zone_assignment
FROM 
    warehouse_employees e
JOIN 
    performance_metrics p ON e.employee_id = p.employee_id
WHERE 
    e.role = 'Picker' 
    AND p.efficiency_rate < 0.85
    AND p.metric_date >= CURRENT_DATE - INTERVAL '7 days'
ORDER BY 
    p.efficiency_rate ASC;
